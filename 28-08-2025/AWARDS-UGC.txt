SELECT TRIM(FIRST_NAME) AS Trimmed_FirstName
FROM Worker;
SELECT UPPER(FIRST_NAME) AS Upper_FirstName
FROM Worker;
SELECT LOWER(DEPARTMENT) AS Lower_Department
FROM Worker;
SELECT 
    TRIM(FIRST_NAME) AS Trimmed_Name,
    UPPER(DEPARTMENT) AS Upper_Dept,
    LOWER(USERNAME) AS Lower_Username
FROM Worker;
SELECT DEPARTMENT, count(department) FROM
Worker GROUP BY DEPARTMENT;
SELECT DEPARTMENT, count(department) FROM
Worker
 GROUP BY DEPARTMENT having count(DEPARTMENT)>3;
 SELECT DEPARTMENT, count(department) FROM
Worker
 GROUP BY DEPARTMENT having sum(DEPARTMENT)>10000000;
SELECT DEPARTMENT, AVG(SALARY) AS Avg_Salary
FROM Worker
GROUP BY DEPARTMENT
HAVING AVG(SALARY) > 100000;
SELECT DEPARTMENT, MAX(SALARY) AS Max_Salary
FROM Worker
GROUP BY DEPARTMENT
HAVING MAX(SALARY) = 500000;
SELECT DEPARTMENT, COUNT(*) AS Employee_Count, SUM(SALARY) AS Total_Salary
FROM Worker
GROUP BY DEPARTMENT
HAVING COUNT(*) > 1 AND SUM(SALARY) > 100000;
SELECT DEPARTMENT, AVG(SALARY) AS Avg_Salary
FROM Worker
GROUP BY DEPARTMENT
HAVING AVG(SALARY) BETWEEN 50000 AND 200000;

CREATE DATABASE UGC_DB;
USE UGC_DB;
CREATE TABLE Worker (
    WORKER_ID INT PRIMARY KEY AUTO_INCREMENT,
    FIRST_NAME VARCHAR(25) NOT NULL,
    LAST_NAME VARCHAR(25) NOT NULL,
    DEPARTMENT VARCHAR(25) NOT NULL,
    SALARY INT NOT NULL
);

CREATE TABLE Title_Awards (
    AWARD_ID INT PRIMARY KEY AUTO_INCREMENT,
    WORKER_ID INT,
    AWARD_TITLE VARCHAR(50) NOT NULL,
    AWARD_YEAR INT,
    FOREIGN KEY (WORKER_ID) REFERENCES Worker(WORKER_ID)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);
