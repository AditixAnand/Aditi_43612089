SELECT MAX(SALARY) AS SecondHighestSalary
FROM Worker
WHERE SALARY < (SELECT MAX(SALARY) FROM Worker);

SELECT DEPARTMENT, SUM(SALARY) AS TotalSalary
FROM Worker
GROUP BY DEPARTMENT;

SELECT FIRST_NAME, LAST_NAME, JOINING_DATE
FROM Worker
WHERE YEAR(JOINING_DATE) = 2014;

SELECT DEPARTMENT, COUNT(*) AS NoOfEmployees
FROM Worker
GROUP BY DEPARTMENT
HAVING COUNT(*) > 2;

SELECT FIRST_NAME, LAST_NAME, SALARY
FROM Worker
WHERE SALARY > (SELECT AVG(SALARY) FROM Worker);

SELECT * FROM Worker ORDER BY AGE ASC LIMIT 1;   

SELECT * FROM Worker ORDER BY AGE DESC LIMIT 1;  

SELECT SALARY, COUNT(*) 
FROM Worker
GROUP BY SALARY
HAVING COUNT(*) > 1;

SELECT FIRST_NAME
FROM Worker
WHERE LEFT(FIRST_NAME,1) = RIGHT(FIRST_NAME,1);

SELECT FIRST_NAME, LAST_NAME, SALARY,
       RANK() OVER (ORDER BY SALARY DESC) AS SalaryRank
FROM Worker;

SELECT DEPARTMENT, AVG(SALARY) AS AvgSalary
FROM Worker
GROUP BY DEPARTMENT
ORDER BY AvgSalary DESC
LIMIT 1;
