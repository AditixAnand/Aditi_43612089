
CREATE DATABASE IF NOT EXISTS school;
USE school;

-- Classes table
CREATE TABLE classes (
    id INT PRIMARY KEY,
    class_name VARCHAR(50)
);

-- Students table
CREATE TABLE students (
    id INT PRIMARY KEY,
    name VARCHAR(50),
    class_id INT,
    FOREIGN KEY (class_id) REFERENCES classes(id)
);
